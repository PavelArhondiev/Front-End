var startingCard = '';
var lastPlanet='';
var lastPlanetFlipDown='';

function cardFlipper(card, planetName) {
    urlString = "url(Images/" + planetName + ".png)";
    document.getElementById(card).dataset.name = planetName;
    var currentPlanet=document.getElementById(card).dataset.name;
    
    document.getElementById(card).style.backgroundImage = urlString;
    document.getElementById(card).classList.remove('faceDown');

if(lastPlanet==currentPlanet ||lastPlanet==''){
        document.getElementById(card).style.backgroundImage = urlString;
        document.getElementById(card).classList.remove('faceDown','Cards');
        document.getElementById(card).classList.add('CardsHidden')

        
    }else{
        document.getElementById(lastPlanetFlipDown).style.backgroundImage = "url(Images/CardBack.png";
        document.getElementById(card).style.backgroundImage = "url(Images/CardBack.png";
    }

    lastPlanet = document.getElementById(card).dataset.name;
    lastPlanetFlipDown = document.getElementById(card).id;
}
